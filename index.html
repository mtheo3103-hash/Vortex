function fetchServerIp() { const id = document.getElementById('serverIdInput').value.trim().toLowerCase().replace(/[^a-z0-9]/g, ''); const resultBox = document.getElementById('resultBox'); const hintText = document.getElementById('hintText'); if (!id) { alert("Bitte gib eine gültige Server-ID ein!"); return; } resultBox.style.borderColor = "#ffcc00"; resultBox.style.color = "#ffcc00"; resultBox.innerText = "Suche im Vortex-Netzwerk..."; currentFetchedIp = ""; // Holt die IP aus dem neuen, unblockierten Speicher fetch(`https://api.jsonstorage.net/v1/json/public/${id}`) .then(res => { if (!res.ok) throw new Error("ID nicht gefunden"); return res.json(); }) .then(data => { if (data && data.ip) { currentFetchedIp = data.ip; resultBox.style.borderColor = "#00ff88"; resultBox.style.color = "#00ff88"; resultBox.innerText = data.ip; hintText.style.color = "#00ff88"; hintText.innerText = "Server gefunden! Klicke auf die Box, um die IP zu kopieren. ✅"; } else { throw new Error("Fehlerhaftes Format"); } }) .catch(err => { resultBox.style.borderColor = "#ff4a4a"; resultBox.style.color = "#ff4a4a"; resultBox.innerText = "❌ ID nicht gefunden oder Server offline"; hintText.style.color = "#7c7c8a"; hintText.innerText = "Bitte überprüfe die Schreibweise oder starte den Tunnel im Panel neu."; }); }
